Atualizando os valores totais que estão faltando na tabela

1)Item_pedido irei colocar os valores na coluna de TotalProduto

update 
        item_pedido i
set 
        i.totalproduto = 

(
   select 
	      (i.quantidadeProduto * p.precoProduto) 
   from 
			produto p  
   where 
         i.Produto_CodigoProduto = p.codigoProduto
);
